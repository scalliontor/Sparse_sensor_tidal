\section{Methodology}
\label{sec:method}
% ══════════════════════════════════════════════════════════════

\subsection{Problem Formulation}
\label{sec:problem}

We study the task of \emph{causal full-field forecasting from sparse boundary sensors}. Given a physical domain $\Omega$ governed by the 2D shallow water equations (SWE), we assume:
\begin{itemize}[nosep]
    \item Only $K$ sensors are available, placed on the domain boundary ($K \ll N_x \times N_y$).
    \item Each sensor records water depth $h$ as a time-series from $t=0$ to some observation time $T_\mathrm{obs}$.
    \item The goal is to predict the full field $\eta(x, y, t)$ for all $(x,y) \in \Omega$ and all $t > T_\mathrm{obs}$.
\end{itemize}

This combines three simultaneous challenges:
\begin{enumerate}[nosep]
    \item \textbf{Sparse:} Sensors cover $\leq 0.1\%$ of the spatial domain.
    \item \textbf{Boundary-only:} No interior measurements exist.
    \item \textbf{Causal:} The model may only use past data; future sensor readings are unavailable.
\end{enumerate}

Formally, we seek a learned operator $\mathcal{G}_\theta$:
\begin{equation}
    \mathcal{G}_\theta : \underbrace{\bigl[\mathbf{s}(t)\bigr]_{t=0}^{T_\mathrm{obs}}}_{\text{sensor history}} \;\longmapsto\; \underbrace{\eta(x, y, t)}_{\text{full field}}, \quad t > T_\mathrm{obs}, \;\; (x,y) \in \Omega,
    \label{eq:causal_op}
\end{equation}
where $\mathbf{s}(t) = [h(x_1,y_1,t), \ldots, h(x_K,y_K,t)] \in \mathbb{R}^K$ is the sensor vector at time $t$.

\paragraph{Contrast with reconstruction (non-causal) setting.}
Prior work on partial-observation DeepONet~\cite{lu2021deeponet} typically assumes access to the complete time-series $\mathbf{s}(t)$ for all $t \in [0, T]$, and predicts $\eta(x,y,t)$ at any queried time---including $t$ that has already been observed. In this setting the branch network sees the entire temporal history. We call this \emph{non-causal reconstruction}. Our causal forecasting formulation is strictly harder: the model receives a prefix of the time-series and must extrapolate both spatially (from boundary to interior) and temporally (from past to future).


\subsection{Governing Equations}
\label{sec:governing}

The physical system is governed by the 2D shallow water equations in conservative form:
\begin{align}
  \frac{\partial h}{\partial t} + \frac{\partial (hu)}{\partial x} + \frac{\partial (hv)}{\partial y} &= 0, \label{eq:mass} \\[4pt]
  \frac{\partial (hu)}{\partial t} + \frac{\partial}{\partial x}\!\left(hu^2 + \frac{1}{2}g h^2\right) + \frac{\partial (huv)}{\partial y} &= 0, \label{eq:momx} \\[4pt]
  \frac{\partial (hv)}{\partial t} + \frac{\partial (huv)}{\partial x} + \frac{\partial}{\partial y}\!\left(hv^2 + \frac{1}{2}g h^2\right) &= 0, \label{eq:momy}
\end{align}
where $h(x,y,t)$ is water depth, $u,v$ are depth-averaged velocities, and $g = 9.81\,\text{m/s}^2$. This system is used as the foundational benchmark in PDEBench~\cite{takamoto2022pdebench}, where each simulation is a radial dam-break with different initial conditions.


\subsection{Baseline: Non-Causal Boundary-DeepONet}
\label{sec:baseline_deeponet}

Our baseline is a standard DeepONet~\cite{lu2021deeponet} operating in the \emph{non-causal reconstruction} regime:

\textbf{Branch Network} (MLP). Receives the flattened sensor time-series across the full temporal window:
\begin{equation}
    \mathbf{u}_\mathrm{branch} = \mathrm{flatten}\bigl[\mathbf{s}(t_0), \ldots, \mathbf{s}(t_T)\bigr] \in \mathbb{R}^{K \cdot N_t}.
\end{equation}
With $K=16$ sensors and $N_t=101$ timesteps: $d_\mathrm{in} = 1{,}616$. Note that the branch observes the future (non-causal).

\textbf{Trunk Network} (MLP). Encodes query location $(\tilde{x}, \tilde{y}, \tilde{t}) \in [-1,1]^3$:
\begin{equation}
    \mathbf{t} = \mathcal{T}(\tilde{x}, \tilde{y}, \tilde{t}) \in \mathbb{R}^{p}.
\end{equation}

\textbf{Output:}
\begin{equation}
    \hat{\eta}(x,y,t) = \sum_{i=1}^{p} b_i \cdot t_i + b_0.
    \label{eq:deeponet_output}
\end{equation}

Architecture: 4-layer MLPs, width 256, GELU activation, latent $p = 128$.


\subsection{ForecastDeepONet: Causal Architecture}
\label{sec:forecast_deeponet}

We propose \textbf{ForecastDeepONet}, modifying two key components of the baseline to achieve causal forecasting capability.

\subsubsection{Causal LSTM Branch}

We replace the MLP branch with an LSTM encoder that processes the sensor time-series \emph{sequentially} from $t=0$ to $t=T_\mathrm{obs}$:
\begin{equation}
    \mathbf{h}_T, \mathbf{c}_T = \mathrm{LSTM}\bigl(\mathbf{s}(t_0), \ldots, \mathbf{s}(t_{T_\mathrm{obs}}})\bigr),
\end{equation}
\begin{equation}
    \mathbf{b} = W_\mathrm{proj} \cdot \mathbf{h}_T^{(\text{top})} \in \mathbb{R}^p,
\end{equation}
where $\mathbf{h}_T^{(\text{top})}$ is the hidden state of the final LSTM layer at the last timestep. By construction, the LSTM output at time $T_\mathrm{obs}$ depends only on inputs $t \leq T_\mathrm{obs}$, enforcing strict causality without any architectural leakage.

Architecture: $\mathrm{LSTM}(\text{input\_size}=K, \text{hidden}=256, \text{layers}=2)$ followed by a linear projection to $\mathbb{R}^p$.

\subsubsection{Fourier Positional Encoding for Trunk}
\label{sec:fourier_pe}

A standard MLP trunk receiving only 3 scalar coordinates $(x,y,t)$ struggles to represent high-frequency spatial patterns~\cite{mildenhall2020nerf}. We apply Fourier positional encoding before the trunk MLP:
\begin{equation}
    \phi(x, y, t) = \Bigl[x,\, y,\, t,\; \sin(\pi \cdot 2^k \cdot x),\, \cos(\pi \cdot 2^k \cdot x),\; \ldots\Bigr]_{k=0}^{L-1},
\end{equation}
expanding $\mathbb{R}^3 \to \mathbb{R}^{3 + 3 \times L \times 2}$. With $L=8$ frequency bands, the trunk input becomes 51-dimensional.

\paragraph{Distinction from Fourier Neural Operator (FNO).}
This encoding is \emph{not} FNO~\cite{li2021fno}. FNO performs actual FFT on the full spatial grid and requires dense field-to-field input---incompatible with sparse sensors. Our Fourier PE is a fixed, input-level encoding of point coordinates that helps the MLP represent high-frequency basis functions, fully compatible with the sparse-to-field DeepONet paradigm.

\subsection{Training Strategy}
\label{sec:training}

\subsubsection{Random Horizon Sampling}
At each training step, we sample $T_\mathrm{obs} \sim \mathrm{Uniform}(T_\mathrm{min}, T_\mathrm{max})$. The branch receives sensor history $\mathbf{s}(0:T_\mathrm{obs})$, while the trunk samples query points only from $t > T_\mathrm{obs}$. This exposes the model to all possible observation lengths during training, ensuring robustness across forecast horizons at inference.

\subsubsection{Loss Function}
Mean squared error between predicted and reference values at randomly sampled future query points:
\begin{equation}
    \mathcal{L} = \frac{1}{M}\sum_{m=1}^{M}\bigl(\hat{\eta}(x_m, y_m, t_m) - \eta^{\star}(x_m, y_m, t_m)\bigr)^2, \quad t_m > T_\mathrm{obs}.
\end{equation}

\subsubsection{Optimization}
AdamW~\cite{loshchilov2019adamw} ($\alpha = 10^{-3}$, weight decay $10^{-4}$), gradient clipping (max norm 1.0), 100 epochs on a single NVIDIA L40S GPU.

\section{Experiments}
\label{sec:experiments}
% ══════════════════════════════════════════════════════════════

\subsection{Benchmark: PDEBench 2D Shallow Water}
\label{sec:pdebench}

\subsubsection{Dataset}
We evaluate on the PDEBench~\cite{takamoto2022pdebench} 2D SWE dataset (Radial Dam Break). The dataset contains $N = 1{,}000$ simulations on a $128 \times 128$ grid over 101 timesteps. We use 800 for training and 200 for validation. Sensors are 16 points placed on the four domain edges (4 per edge), covering 16/16{,}384 = \textbf{0.098\%} of the grid.

\subsubsection{Evaluation Metric}
Relative L2 error averaged over all validation samples:
\begin{equation}
    \varepsilon_{\text{Rel-L2}} = \frac{\|\hat{\eta} - \eta^{\star}\|_2}{\|\eta^{\star}\|_2}.
\end{equation}

\subsubsection{Results: Non-Causal Baselines}

We first establish the performance ceiling with non-causal models that observe the full temporal window:

\begin{table}[H]
\centering

\label{tab:baselines}
\vspace{4pt}
\begin{tabular}{llcc}
\toprule
\textbf{Model} & \textbf{Observation} & \textbf{Rel.\ L2 (\%)} $\downarrow$ & \textbf{Source} \\
\midrule
U-Net & Full $128 \times 128$ field & 12.80 & \cite{takamoto2022pdebench} \\

We now evaluate ForecastDeepONet, which operates under the strictly harder causal constraint: only past data, predict future.

\paragraph{Key findings.}
\begin{enumerate}[nosep]
    \item \textbf{Causal outperforms non-causal.} ForecastDeepONet (4.46\%, causal) surpasses Boundary-DeepONet (5.25\%, non-causal) despite having strictly less information. The LSTM extracts temporal dynamics more effectively than an MLP processing a flattened time-series.
    \item \textbf{No forecast degradation.} Error remains stable (4.33--4.72\%) across all horizons, from 21 to 81 future timesteps. The model does not degrade when asked to predict further into the future.
    \item \textbf{No overfitting.} Train Rel L2 = 4.38\%, Val Rel L2 = 4.36\% (gap = $-$0.02\%). Training completes in 174 seconds for 100 epochs.
\end{enumerate}

\subsection{Supporting Validation: Gulf of Tonkin Reconstruction}
\label{sec:tonkin}
% ──────────────────────────────────────────────────────────────

To validate the broader operator learning framework on a real-world domain, we apply the Fourier DeepONet (non-causal, 20 ocean sensors) to reconstruct tidal propagation in \vn{Vịnh Bắc Bộ} (Gulf of Tonkin), Vietnam.

\subsubsection{Setup}
\begin{itemize}[nosep]
    \item \textbf{Domain:} $130 \times 110$ grid over real GEBCO bathymetry (0--1{,}500\,m depth).
    \item \textbf{Solver:} Well-balanced HLL with Hydrostatic Reconstruction.
    \item \textbf{Forcing:} Synthetic tidal constituents (K1, O1, P1, M2, S2; total $\pm$1.3\,m), consistent with the diurnal-dominant character of the Gulf.
    \item \textbf{Sensors:} 20 ocean sensors (bath $< 0$\,m) along south and east boundaries.
    \item \textbf{Dataset:} 20 simulation windows of 48\,h each. Stratified train/test split by tidal regime (neap/medium/spring). Ocean-only sampling (7{,}645 cells).
\end{itemize}

\subsubsection{Results}

\begin{table}[H]
\centering
\caption{Gulf of Tonkin reconstruction: progressive improvement through systematic debugging.}
\label{tab:tonkin}
\vspace{4pt}
\begin{tabular}{llcc}
\toprule
\textbf{Configuration} & \textbf{Key change} & \textbf{RMSE (cm)} & \textbf{Rel.\ L2 (\%)} \\
\midrule
Original setup & Copernicus SLA, land sensors & $\sim$50 & 93.6 \\
+ Physics fixes & Tidal forcing + ocean sensors + mask & 6.45 & 27.1 \\
\textbf{+ Fourier DeepONet} & \textbf{Fourier PE + latent 256} & \textbf{3.54} & \textbf{14.8} \\
\bottomrule
\end{tabular}
\end{table}

The 93.6\% $\to$ 14.8\% improvement was achieved through systematic 4-layer debugging (physics/forcing/sensors/dataset), detailed in the supplementary material. This case study serves as engineering validation that the operator-learning framework generalizes beyond idealized benchmarks to real bathymetric domains.
